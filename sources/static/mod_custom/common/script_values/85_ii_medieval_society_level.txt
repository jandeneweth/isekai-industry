# Medieval society level is a score from 0-100, used to determine modifiers to the country
# It is calculated from the country's laws.
medieval_society_level = {
	value = 0

    # Army model
    if = {
        limit = {
            OR = {
                has_law = law_type:law_peasant_levies
            }
        }
        add = 10
    }

    # Bureaucracy
    if = {
        limit = {
            OR = {
                has_law = law_type:law_hereditary_bureaucrats
            }
        }
        add = 10
    }

    # Church and state
    if = {
        limit = {
            OR = {
                has_law = law_type:law_state_religion
            }
        }
        add = 10
    }

    # Distribution of power
    if = {
        limit = {
            OR = {
                has_law = law_type:law_autocracy
                has_law = law_type:law_elder_council
            }
        }
        add = 10
    }
    else_if = {
        limit = {
            OR = {
                has_law = law_type:law_oligarchy
                has_law = law_type:law_landed_voting
            }
        }
        add = 5

    }

    # Economic system
    if = {
        limit = {
            OR = {
                has_law = law_type:law_traditionalism
                has_law = law_type:law_industry_banned
            }
        }
        add = 10
    }
    else_if = {
        limit = {
            OR = {
                has_law = law_type:law_agrarianism
            }
        }
        add = 5
    }

    # Governance principles
    if = {
        limit = {
            OR = {
                has_law = law_type:law_chiefdom
                has_law = law_type:law_monarchy
                has_law = law_type:law_theocracy
            }
        }
        add = 10
    }

    # Land reform
    if = {
        limit = {
            OR = {
                has_law = law_type:law_serfdom
            }
        }
        add = 10
    }
    else_if = {
        limit = {
            OR = {
                has_law = law_type:law_tenant_farmers
            }
        }
        add = 5
    }

    # Migration
    if = {
        limit = {
            OR = {
                has_law = law_type:law_closed_borders
            }
        }
        add = 10
    }

    # Taxation
    if = {
        limit = {
            OR = {
                has_law = law_type:law_consumption_based_taxation
                has_law = law_type:law_land_based_taxation
            }
        }
        add = 10
    }
    else_if = {
        limit = {
            OR = {
                has_law = law_type:law_per_capita_based_taxation
            }
        }
        add = 5
    }

    # Trade policy
    if = {
        limit = {
            OR = {
                has_law = law_type:law_isolationism
                has_law = law_type:law_city_tolls
            }
        }
        add = 10
    }
    else_if = {
        limit = {
            OR = {
                has_law = law_type:law_mercantilism
            }
        }
        add = 5
    }
}
